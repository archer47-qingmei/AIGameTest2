### Task 1: 扩展 CombatEngine 和 CombatScreen 支持 4 个新 action type

新增 4 个 action type：
- `discard_curse` — 往弃牌堆塞 N 张心魔（幻蛾用）
- `attack_weak` — 攻击 + 对玩家施加 1 层虚弱（毒涎草用）
- `attack_vulnerable` — 攻击 + 对玩家施加 1 层易伤（毒涎草用）
- `charge_attack` — 机制同 attack，意图显示为"蓄力攻击"（石妖第 4 回合用）

**Files:**
- Modify: `CombatEngine.gd`
- Modify: `combat/CombatScreen.gd`

- [ ] **Step 1: 修改 `CombatEngine.gd` — `_do_enemy_turn()` 的 match 块**

找到 `_do_enemy_turn()` 中的 `match action.type:` 块（当前只有 "attack"、"poison"、默认 block 三个分支），将整个 match 块替换为：

```gdscript
match action.type:
    "attack", "charge_attack":
        var hp_before: int = player.hp
        EffectResolver.apply_damage(enemies[i], player, action.value)
        var dmg: int = hp_before - player.hp
        if dmg > 0:
            player_damaged.emit(dmg)
        enemies[i].weak = max(0, enemies[i].weak - 1)
    "attack_weak":
        var hp_before: int = player.hp
        EffectResolver.apply_damage(enemies[i], player, action.value)
        var dmg: int = hp_before - player.hp
        if dmg > 0:
            player_damaged.emit(dmg)
        player.add_weak(1)
        enemies[i].weak = max(0, enemies[i].weak - 1)
    "attack_vulnerable":
        var hp_before: int = player.hp
        EffectResolver.apply_damage(enemies[i], player, action.value)
        var dmg: int = hp_before - player.hp
        if dmg > 0:
            player_damaged.emit(dmg)
        player.add_vulnerable(1)
        enemies[i].weak = max(0, enemies[i].weak - 1)
    "poison":
        for _j in action.value:
            _draw_pile.append(VENOM_CARD.duplicate())
        _draw_pile.shuffle()
    "discard_curse":
        for _j in action.value:
            _discard_pile.append(VENOM_CARD.duplicate())
    _:
        enemies[i].add_block(action.value)
```

- [ ] **Step 2: 修改 `combat/CombatScreen.gd` — `_intent_text()` 方法**

找到 `_intent_text()` 方法（当前有 attack/poison/默认三个分支），将整个方法替换为：

```gdscript
func _intent_text(action: EnemyActionData, e: Combatant) -> String:
    var val: int = int(action.value * 0.75) if e.weak > 0 else action.value
    match action.type:
        "attack":
            return "意图：攻击 %d" % val
        "charge_attack":
            return "意图：蓄力攻击 %d" % val
        "attack_weak":
            return "意图：攻击 %d + 虚弱" % val
        "attack_vulnerable":
            return "意图：攻击 %d + 易伤" % val
        "poison":
            return "意图：投毒 %d" % action.value
        "discard_curse":
            return "意图：心魔 ×%d" % action.value
        _:
            return "意图：格挡 %d" % action.value
```
