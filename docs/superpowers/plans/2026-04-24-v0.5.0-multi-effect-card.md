## Task 1：新增 slash.tres 并更新 CombatEngine

**Files:**
- Create: `e:/MyWork/AIGameTest2/data/cards/slash.tres`
- Modify: `e:/MyWork/AIGameTest2/CombatEngine.gd`

> **背景：** CardData.gd UID = `uid://13mprie5g221`，CardEffectData.gd UID = `uid://cxv8nrm4tp7dk`（来自各自的 .uid 文件）。slash.tres 通过 sub_resource 内嵌两条效果记录。load_steps = 2 ext + 2 sub + 1 root = 5。

- [ ] **Step 1：创建 data/cards/slash.tres**

创建 `e:/MyWork/AIGameTest2/data/cards/slash.tres`：

```
[gd_resource type="Resource" script_class="CardData" load_steps=5 format=3]

[ext_resource type="Script" uid="uid://13mprie5g221" path="res://CardData.gd" id="1_CardData"]
[ext_resource type="Script" uid="uid://cxv8nrm4tp7dk" path="res://CardEffectData.gd" id="2_CardEffectData"]

[sub_resource type="Resource" id="Effect_damage"]
script = ExtResource("2_CardEffectData")
type = "damage"
value = 4

[sub_resource type="Resource" id="Effect_block"]
script = ExtResource("2_CardEffectData")
type = "block"
value = 4

[resource]
script = ExtResource("1_CardData")
card_name = "斩击"
cost = 2
effects = [SubResource("Effect_damage"), SubResource("Effect_block")]
```

- [ ] **Step 2：更新 CombatEngine.gd 的 setup()**

将 `e:/MyWork/AIGameTest2/CombatEngine.gd` 第 31–38 行（卡牌加载区）替换为：

```gdscript
	var strike: CardData = preload("res://data/cards/strike.tres")
	var defend: CardData = preload("res://data/cards/defend.tres")
	var bash: CardData = preload("res://data/cards/bash.tres")
	var slash: CardData = preload("res://data/cards/slash.tres")
	for i in 4:
		_draw_pile.append(strike.duplicate())
	for i in 4:
		_draw_pile.append(defend.duplicate())
	_draw_pile.append(bash.duplicate())
	_draw_pile.append(slash.duplicate())
```

改完后 `setup()` 完整内容应为：

```gdscript
func setup() -> void:
	_enemy_data = preload("res://data/enemies/jaw_worm.tres")
	enemy = Combatant.new()
	enemy.display_name = _enemy_data.display_name
	enemy.hp = _enemy_data.hp
	enemy.max_hp = _enemy_data.hp
	enemy.block = 0

	player = Combatant.new()
	player.display_name = "玩家"
	player.hp = 70
	player.max_hp = 70
	player.block = 0

	var strike: CardData = preload("res://data/cards/strike.tres")
	var defend: CardData = preload("res://data/cards/defend.tres")
	var bash: CardData = preload("res://data/cards/bash.tres")
	var slash: CardData = preload("res://data/cards/slash.tres")
	for i in 4:
		_draw_pile.append(strike.duplicate())
	for i in 4:
		_draw_pile.append(defend.duplicate())
	_draw_pile.append(bash.duplicate())
	_draw_pile.append(slash.duplicate())

	_draw_pile.shuffle()
	_start_player_turn()
```
