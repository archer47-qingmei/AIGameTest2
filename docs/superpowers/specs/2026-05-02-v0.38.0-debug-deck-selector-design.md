# v0.38.0 调试牌组选择器设计

**目标：** 在角色选择界面新增"测试模式"入口，允许开发者自由勾选初始牌组中的卡牌（支持多份），以空牌组为基础开始测试跑图。

---

## 流程

```
CharSelectScreen
  ├── 剑修 → GameManager.start_new_run("sword")（不变）
  └── 测试模式（新按钮）→ DebugDeckScreen
                              ├── 开始测试 → GameManager.start_debug_run(cards)
                              └── 返回 → CharSelectScreen
```

---

## 数据层

无新增 `.tres` 文件。`DebugDeckScreen` 直接从 `CardPool.SWORD_REWARD_CARDS` 读取全部 21 张牌路径并动态加载。

---

## 逻辑层

### GameManager.gd

新增方法 `start_debug_run(cards: Array[CardData])`：

```gdscript
func start_debug_run(cards: Array[CardData]) -> void:
    pending_card_reward = true
    player_state = PlayerState.new()
    player_state.character = "sword"
    for path in CardPool.SWORD_START_RELICS:
        player_state.relics.append((load(path) as RelicData).duplicate())
    for card in cards:
        player_state.deck.append(card.duplicate())
    var nodes: Array[NodeData] = MapGenerator.generate()
    player_state.map_all_nodes.assign(nodes)
    var start_nodes: Array[NodeData] = []
    for nd in nodes:
        if nd.config.column == 0:
            start_nodes.append(nd)
    player_state.available_nodes.assign(start_nodes)
    current_phase = Phase.MAP
    get_tree().change_scene_to_file("res://map/MapScreen.tscn")
```

与 `start_new_run` 的唯一区别：跳过硬编码起始牌组，改用传入的 `cards` 数组。角色固定 "sword"，遗物正常分配。

新增方法 `go_to_debug_deck()`：

```gdscript
func go_to_debug_deck() -> void:
    current_phase = Phase.CHAR_SELECT
    get_tree().change_scene_to_file("res://debug/DebugDeckScreen.tscn")
```

新增方法 `go_to_char_select()`（供 DebugDeckScreen 返回使用）：

```gdscript
func go_to_char_select() -> void:
    current_phase = Phase.CHAR_SELECT
    get_tree().change_scene_to_file("res://char_select/CharSelectScreen.tscn")
```

### DebugDeckScreen.tscn

```
Control（全屏）
└── VBoxContainer（居中，宽 520px，高 560px）
    ├── LblTitle: "调试牌组"
    ├── ScrollContainer（fill height）
    │   └── CardListContainer: VBoxContainer（动态生成）
    ├── LblSelected: "已选：0 张"
    └── HBoxContainer
        ├── BtnBack: "返回"
        └── BtnStart: "开始测试"（已选=0时禁用）
```

每行（动态生成的 HBoxContainer）：

```
[牌名（80px）] [效果描述（自动换行，280px）] [BtnMinus "−"（30px）] [LblCount "0"（30px）] [BtnPlus "+"（30px）]
```

### DebugDeckScreen.gd

```gdscript
var _card_paths: Array[String] = []
var _cards: Array[CardData] = []
var _counts: Array[int] = []   # 与 _cards 下标一一对应

func _ready() -> void:
    _card_paths.assign(CardPool.SWORD_REWARD_CARDS)
    for path in _card_paths:
        _cards.append(load(path) as CardData)
        _counts.append(0)
    _build_rows()
    _refresh()

func _build_rows() -> void:
    for i in _cards.size():
        var row := HBoxContainer.new()
        var lbl_name := Label.new()
        lbl_name.text = _cards[i].card_name
        lbl_name.custom_minimum_size.x = 80
        var lbl_desc := Label.new()
        lbl_desc.text = _cards[i].get_description()
        lbl_desc.custom_minimum_size.x = 280
        lbl_desc.autowrap_mode = TextServer.AUTOWRAP_WORD_SMART
        var btn_minus := Button.new()
        btn_minus.text = "−"
        btn_minus.custom_minimum_size.x = 30
        btn_minus.pressed.connect(_on_minus.bind(i))
        var lbl_count := Label.new()
        lbl_count.text = "0"
        lbl_count.custom_minimum_size.x = 30
        lbl_count.horizontal_alignment = HORIZONTAL_ALIGNMENT_CENTER
        var btn_plus := Button.new()
        btn_plus.text = "+"
        btn_plus.custom_minimum_size.x = 30
        btn_plus.pressed.connect(_on_plus.bind(i))
        row.add_child(lbl_name)
        row.add_child(lbl_desc)
        row.add_child(btn_minus)
        row.add_child(lbl_count)
        row.add_child(btn_plus)
        %CardListContainer.add_child(row)

func _on_minus(i: int) -> void:
    _counts[i] = max(0, _counts[i] - 1)
    _refresh()

func _on_plus(i: int) -> void:
    _counts[i] = min(5, _counts[i] + 1)
    _refresh()

func _refresh() -> void:
    var total := 0
    for i in _cards.size():
        total += _counts[i]
        var row: HBoxContainer = %CardListContainer.get_child(i)
        row.get_child(3).text = str(_counts[i])   # LblCount
    %LblSelected.text = "已选：%d 张" % total
    %BtnStart.disabled = total == 0

func _on_start_pressed() -> void:
    var cards: Array[CardData] = []
    for i in _cards.size():
        for _j in _counts[i]:
            cards.append(_cards[i])
    GameManager.start_debug_run(cards)

func _on_back_pressed() -> void:
    GameManager.go_to_char_select()
```

`_refresh()` 中通过行内子节点下标访问 LblCount（下标 3）和更新禁用状态；行结构固定，下标稳定。
